#Sự khác biệt giữa thẻ block và inline:
- Thẻ inline có độ rộng bằng nội dung mà nó chứa, nếu có nhiều thẻ inline những thẻ inline nằm cùng trên một hàng, bị hạn chế css
- Thẻ block có độ rộng full  phần tử chứa nó,  có nhiều thẻ block thì sẽ rớt xuống hàng.


#Sự khác biệt class và id:
- class : có thể trùng nhau, sử dụng cho nhiều phần tử giống nhau, có thể đặt nhiều tên .btn
- id : là duy nhất , #btn

#BEM block-element-modifier
block__element--modifier
block--modifier
block__element
block: laptop 
element: mouse, keyboard
modifier: big, small, blue

#Thực hành BEM
block: flash-message
element: image, title, desc, button
modifier: primary, secondary
Ex: 
flash-message__image, flash-message__title, flash-message__desc
flash-message__title--secondary, flash-message__title--primary

#Ví dụ thẻ 
block: div, section, body, h1, h2, h3, h4, h5, h6, article
inline: span, img, strong, em, i, b, a, button, sub, sup
semantic: section, aside, main, article, nav, footer
